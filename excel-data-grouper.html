Parallel Excel Data Grouper

Python Pandas Threading Tkinter

A multi-threaded desktop application that processes and aggregates multiple Excel files in parallel, featuring a real-time console output GUI.

Project Overview

Processing multiple Excel files sequentially can be a bottleneck. This tool utilizes Python's concurrent.futures.ThreadPoolExecutor to read and process multiple Excel files simultaneously, significantly reducing execution time.

Key Engineering Features:

  • Parallel Processing: Reads multiple Excel files concurrently using thread pools.
  • Custom Console GUI: Redirects standard output (sys.stdout) to a Tkinter ScrolledText widget for real-time logging.
  • Automated Aggregation: Concatenates data from all sheets and files, performs GroupBy operations, and exports the summarized data.
import os
import glob
import pandas as pd
import tkinter as tk
from tkinter import filedialog, scrolledtext
import sys
import threading
from concurrent.futures import ThreadPoolExecutor, as_completed


class ConsoleRedirector:
    """A helper class to redirect stdout to a tkinter Text widget."""
    def __init__(self, text_widget):
        self.text_widget = text_widget

    def write(self, string):
        self.text_widget.insert(tk.END, string)
        self.text_widget.see(tk.END)
        self.text_widget.update_idletasks()

    def flush(self):
        pass


def read_single_excel_file(file_path):
    """
    Reads all sheets from a single Excel file.
    Designed to be run in a separate thread.
    """
    try:
        sheet_names = pd.ExcelFile(file_path).sheet_names
        print(f"  -> Reading {os.path.basename(file_path)}...")
        file_dataframes = []
        for sheet_name in sheet_names:
            df = pd.read_excel(file_path, sheet_name=sheet_name, header=None)
            if not df.empty:
                file_dataframes.append(df.iloc[1:])  # Skip header row
        return file_dataframes
    except Exception as e:
        print(f"      [Warning] Could not process {os.path.basename(file_path)}. Error: {e}")
        return []


def group_and_sum_excel_files():
    """
    Prompts user to select a folder, processes all Excel files in parallel,
    groups and sums data, and saves the result.
    """
    # --- 1. Select a folder ---
    print("Please select the folder containing your Excel files.")
    folder_path = filedialog.askdirectory(title="Select Folder With Excel Files")

    if not folder_path:
        print("No folder selected. Aborting.")
        return

    print(f"Processing files in: {folder_path}")
    excel_files = glob.glob(os.path.join(folder_path, "*.xls*"))

    if not excel_files:
        print("No Excel files found in the selected folder. Exiting.")
        return

    # --- 2. Process all files and sheets in parallel ---
    all_data_from_all_sheets = []
    print("\nStarting parallel file reading...")

    with ThreadPoolExecutor() as executor:
        future_to_file = {executor.submit(read_single_excel_file, f): f for f in excel_files}

        for future in as_completed(future_to_file):
            try:
                dataframes_from_file = future.result()
                if dataframes_from_file:
                    all_data_from_all_sheets.extend(dataframes_from_file)
            except Exception as e:
                file = future_to_file[future]
                print(f"      [Error] Exception while processing {os.path.basename(file)}: {e}")

    if not all_data_from_all_sheets:
        print("No data could be read from the files. Exiting.")
        return

    print("\nCombining all data...")
    combined_df = pd.concat(all_data_from_all_sheets, ignore_index=True)

    # --- 3. Group and Sum the Data ---
    print("Grouping and summing data...")
    try:
        # Column indices to group by (0-based, based on source file structure)
        group_by_columns = [0, 1, 2, 3, 4, 6, 17]
        numeric_col_a = 7    # First numeric column to sum
        numeric_col_b = 12   # Second numeric column to sum

        combined_df[numeric_col_a] = pd.to_numeric(combined_df[numeric_col_a], errors='coerce')
        combined_df[numeric_col_b] = pd.to_numeric(combined_df[numeric_col_b], errors='coerce')

        grouped_df = combined_df.groupby(group_by_columns).agg(
            metric_a_sum=(numeric_col_a, 'sum'),
            metric_b_sum=(numeric_col_b, 'sum')
        ).reset_index()

        grouped_df.columns = [
            "col_a", "col_b", "col_c", "col_d",
            "col_e", "col_f", "col_g",
            "metric_a", "metric_b"
        ]
    except Exception as e:
        print(f"\n[Error] An error occurred during grouping.")
        print(f"Ensure your files have the correct structure (at least 18 columns).")
        print(f"Details: {e}")
        return

    # --- 4. Write the results to a new Excel file ---
    print("Writing results to Excel file...")
    desktop_path = os.path.join(os.path.expanduser("~"), "Desktop")
    output_filepath = os.path.join(desktop_path, "Grouped_Data_Output.xlsx")

    max_rows_per_sheet = 1048575
    num_sheets = (len(grouped_df) - 1) // max_rows_per_sheet + 1

    with pd.ExcelWriter(output_filepath, engine='xlsxwriter') as writer:
        for i in range(num_sheets):
            sheet_name = "All_Grouped" if i == 0 else f"All_Grouped_{i + 1}"
            start_row = i * max_rows_per_sheet
            chunk_df = grouped_df.iloc[start_row: start_row + max_rows_per_sheet]
            chunk_df.to_excel(writer, sheet_name=sheet_name, index=False)
            print(f"  -> Wrote {len(chunk_df)} rows to sheet '{sheet_name}'.")

    print(f"\nSuccess! All files have been processed.")
    print(f"Results saved to: {output_filepath}")


if __name__ == "__main__":
    root = tk.Tk()
    root.title("Excel Data Grouper Console")
    root.geometry("800x600")

    console_frame = tk.Frame(root, padx=10, pady=10)
    console_frame.pack(fill=tk.BOTH, expand=True)

    console_widget = scrolledtext.ScrolledText(
        console_frame,
        wrap=tk.WORD,
        bg="#1e1e1e",
        fg="#d4d4d4",
        font=("Consolas", 10)
    )
    console_widget.pack(fill=tk.BOTH, expand=True)

    sys.stdout = ConsoleRedirector(console_widget)

    def processing_logic():
        try:
            group_and_sum_excel_files()
        except Exception as e:
            print(f"\n--- A CRITICAL ERROR OCCURRED ---")
            print(f"Error details: {e}")
        finally:
            start_button.config(state=tk.NORMAL, text="Select Folder and Start Again")
            print("\n--- PROCESSING COMPLETE ---")

    def start_processing():
        start_button.config(state=tk.DISABLED, text="Processing...")
        thread = threading.Thread(target=processing_logic, daemon=True)
        thread.start()

    button_frame = tk.Frame(root, pady=10)
    button_frame.pack(fill=tk.X)
    start_button = tk.Button(
        button_frame,
        text="Select Folder and Start Processing",
        command=start_processing,
        font=("Segoe UI", 12, "bold"),
        bg="#4CAF50",
        fg="white",
        padx=10,
        pady=5
    )
    start_button.pack()

    print("Welcome to the Excel Data Grouper!")
    print("Click the button below to begin.\n")

    root.mainloop()
