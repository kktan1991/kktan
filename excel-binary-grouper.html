Excel Binary Sheet Grouper

Python Pandas Tkinter Data Processing

A Python automation tool that processes massive Excel Binary (.xlsb) files, extracting, grouping, and aggregating data across multiple sheets with real-time progress tracking.

Project Overview

Processing large Excel Binary files (.xlsb) with numerous sheets can be incredibly slow and memory-intensive. This tool automates the extraction and aggregation of specific data columns across all numbered sheets within a workbook.

Key Engineering Features:

  • Binary File Processing: Utilizes the pyxlsb engine in Pandas to efficiently read large binary Excel files.
  • Automated Aggregation: Automatically identifies numbered sheets, extracts specific columns, and performs GroupBy sum operations.
  • Real-Time ETA: Features a Tkinter GUI with a progress bar and dynamic ETA calculation to keep users informed during long processing jobs.
import pandas as pd
import tkinter as tk
import time
import os
from tkinter import filedialog, messagebox, ttk

def process_excel(file_path, progress_bar, time_label):
    xls = pd.ExcelFile(file_path, engine='pyxlsb')
    numbered_sheets = [s for s in xls.sheet_names if s.isdigit()]
    all_dfs = []
    total = len(numbered_sheets)
    skipped = []

    # Get header names from first numbered sheet
    first_numbered = numbered_sheets[0]
    first_df_header = pd.read_excel(file_path, sheet_name=first_numbered, engine='pyxlsb', nrows=0)
    header_names_from_first_numbered = first_df_header.columns.tolist()

    # Update these indexes for your columns: 2=C, 9=J, 5=F, 11=L, 6=G
    col_indices = [2, 9, 5, 11, 6]
    start_time = time.time()

    for idx, sheet in enumerate(numbered_sheets, 1):
        df = pd.read_excel(file_path, sheet_name=sheet, engine='pyxlsb', header=None)
        if df.shape[1] <= max(col_indices):
            skipped.append(str(sheet))
            continue
        try:
            # Assign output headers from first numbered sheet
            df.columns = header_names_from_first_numbered + [f'Extra_{i}' for i in range(df.shape[1] - len(header_names_from_first_numbered))]
            group_cols = [df.columns[i] for i in col_indices[:-1]]
            sum_col = df.columns[col_indices[-1]]
        except IndexError:
            skipped.append(str(sheet))
            continue

        grouped = df.groupby(group_cols, as_index=False)[sum_col].sum()
        grouped['sheet'] = sheet
        all_dfs.append(grouped)

        # Progress & ETA
        prog = int(idx / total * 100)
        progress_bar['value'] = prog
        elapsed = time.time() - start_time
        avg_per_sheet = elapsed / idx
        remaining = avg_per_sheet * (total - idx)
        mins, secs = divmod(int(remaining), 60)
        time_label.config(text=f"{prog}% complete | Estimated time left: {mins}m {secs}s")
        root.update_idletasks()

    # Combine and save output
    if all_dfs:
        final_df = pd.concat(all_dfs, ignore_index=True)
        out_headers = [header_names_from_first_numbered[i] for i in col_indices[:-1]] + [header_names_from_first_numbered[col_indices[-1]], 'sheet']
        final_df.columns = out_headers
        
        # Autosave output in same directory with suffix '_grouped.xlsx'
        base, ext = os.path.splitext(file_path)
        output_path = base + "_grouped.xlsx"
        final_df.to_excel(output_path, index=False)
        messagebox.showinfo("Export Successful", f"Output saved to:\n{output_path}")
    else:
        messagebox.showwarning("No Data", "No valid sheets processed.\nSkipped sheets: " + ', '.join(skipped))

    progress_bar['value'] = 100
    time_label.config(text="100% complete | Processing finished.")

def select_file(progress_bar, time_label):
    file_path = filedialog.askopenfilename(title="Select Excel Binary File", filetypes=[("Excel Binary Files", "*.xlsb")])
    if file_path:
        try:
            progress_bar['value'] = 0
            time_label.config(text="")
            process_excel(file_path, progress_bar, time_label)
        except Exception as e:
            messagebox.showerror("Error", f"Processing failed: {e}")
            progress_bar['value'] = 0
            time_label.config(text="")

root = tk.Tk()
root.title("Excel Binary Sheet Grouper (No Display, AutoSave)")

frame = tk.Frame(root)
frame.pack(padx=10, pady=10)

button = tk.Button(frame, text="Select Excel Binary File", command=lambda: select_file(progress_bar, time_label))
button.pack(pady=5)

progress_bar = ttk.Progressbar(root, orient="horizontal", length=500, mode='determinate')
progress_bar.pack(padx=10, pady=5)

time_label = tk.Label(root, text="")
time_label.pack(padx=10, pady=5)

root.mainloop()
