1M+ Row Multi-Sheet Exporter

Python Polars Tkinter Data Engineering

A high-performance desktop tool built with Polars to generate, process, and export massive datasets (1M+ rows) across multiple Excel sheets without memory crashes.

Project Overview

This application solves a common enterprise problem: Excel crashing when trying to export or handle datasets larger than 1 million rows. By leveraging the blazing-fast Polars library, this tool splits massive datasets into manageable chunks and writes them across multiple Excel sheets seamlessly.

Key Engineering Features:

  • Polars Integration: Uses Polars instead of Pandas for significantly faster data generation and processing with lower memory overhead.
  • Chunked Excel Writing: Automatically slices large datasets and writes them to multiple sheets (e.g., 1M rows per sheet) using xlsxwriter.
  • Interactive GUI: A clean Tkinter interface allowing users to set intervals, row limits, and manage the end-to-end pipeline with a single click.
import polars as pl
import numpy as np
import os
import glob
import tkinter as tk
from tkinter import messagebox, filedialog
from datetime import datetime

class DataMasterPro:
    def __init__(self, root):
        self.root = root
        self.root.title("Polars Multi-Sheet Master")
        self.root.geometry("500x520")
        
        self.base_path = os.getcwd()
        self.folder_name = "data_input"
        self.full_excel_file = "MASTER_DATA_MULTISHEET.xlsx" 
        
        self.setup_ui()

    def setup_ui(self):
        main_frame = tk.Frame(self.root, padx=20, pady=20)
        main_frame.pack(expand=True, fill="both")

        tk.Label(main_frame, text="1M+ Row Multi-Sheet Exporter", font=('Segoe UI', 14, 'bold')).pack(pady=5)
        
        # Path Selection
        path_frame = tk.LabelFrame(main_frame, text="Working Directory", padx=10, pady=5)
        path_frame.pack(fill="x", pady=5)
        self.path_var = tk.StringVar(value=self.base_path)
        tk.Entry(path_frame, textvariable=self.path_var, state='readonly').pack(side="left", expand=True, fill="x")
        tk.Button(path_frame, text="Browse", command=self.select_path).pack(side="right", padx=5)

        # Settings
        input_frame = tk.LabelFrame(main_frame, text="Settings", padx=10, pady=10)
        input_frame.pack(fill="x", pady=10)
        tk.Label(input_frame, text="Interval (1m, 1h, 1d):").pack(side="left")
        self.interval_entry = tk.Entry(input_frame, width=8)
        self.interval_entry.insert(0, "1h") 
        self.interval_entry.pack(side="left", padx=5)
        
        tk.Label(input_frame, text="Rows/Sheet:").pack(side="left", padx=(10,0))
        self.row_limit_entry = tk.Entry(input_frame, width=10)
        self.row_limit_entry.insert(0, "1000000") 
        self.row_limit_entry.pack(side="left", padx=5)

        # Buttons
        tk.Button(main_frame, text="Step 1: Clear Old & Generate New Data", height=2, bg="#e1f5fe", command=self.generate_data).pack(fill="x", pady=5)
        tk.Button(main_frame, text="Step 2: Merge to Multi-Sheet Excel", height=2, bg="#e8f5e9", command=self.merge_to_excel).pack(fill="x", pady=5)
        tk.Button(main_frame, text="Step 3: Open Excel File", height=2, bg="#c8e6c9", command=self.open_full_file).pack(fill="x", pady=5)

        self.status_var = tk.StringVar(value="Ready")
        tk.Label(main_frame, textvariable=self.status_var, fg="#555", font=('Segoe UI', 9, 'italic')).pack(pady=10)

    def select_path(self):
        selected = filedialog.askdirectory()
        if selected:
            self.path_var.set(selected)
            self.base_path = selected

    def generate_data(self):
        user_interval = self.interval_entry.get().strip()
        input_dir = os.path.join(self.base_path, self.folder_name)
        try:
            # Clear old files to prevent column name mismatch
            if os.path.exists(input_dir):
                for f in glob.glob(os.path.join(input_dir, "*.csv")):
                    os.remove(f)
            else:
                os.makedirs(input_dir)

            self.status_var.set("Generating...")
            self.root.update()

            for i in range(1, 4):
                num_rows = 400000
                # FIX: Using datetime_range for sub-day intervals
                dates = pl.datetime_range(
                    start=datetime(2023, 1, 1), 
                    end=datetime(2023, 12, 31), 
                    interval=user_interval, 
                    eager=True
                ).sample(num_rows, with_replacement=True)

                df = pl.DataFrame({
                    "ID": np.arange(1, num_rows + 1) + ((i-1) * num_rows),
                    "Timestamp": dates, # Synchronized Column Name
                    "Revenue": np.random.uniform(10, 1000, size=num_rows).round(2),
                    "Status": np.random.choice(["Paid", "Pending"], size=num_rows)
                })
                df.write_csv(os.path.join(input_dir, f"raw_part_{i}.csv"))
            
            self.status_var.set("Success: New data ready.")
            messagebox.showinfo("Success", "Generated 1.2M rows with 'Timestamp' header.")
        except Exception as e: 
            messagebox.showerror("Error", f"Generation failed: {e}")

    def merge_to_excel(self):
        input_dir = os.path.join(self.base_path, self.folder_name)
        output_path = os.path.join(self.base_path, self.full_excel_file)
        
        try:
            # Force import to ensure it's visible
            import xlsxwriter 
            
            files = glob.glob(os.path.join(input_dir, "*.csv"))
            if not files:
                messagebox.showwarning("Empty", "No files found. Run Step 1.")
                return
            
            self.status_var.set("Reading & Merging...")
            self.root.update()
            
            # Combine all data
            full_df = pl.concat([pl.read_csv(f) for f in files])
            total_rows = len(full_df)
            chunk_size = int(self.row_limit_entry.get())
            
            self.status_var.set("Writing Excel Sheets...")
            self.root.update()

            # --- NEW BYPASS LOGIC ---
            # We use a context manager to handle the workbook manually
            with xlsxwriter.Workbook(output_path) as workbook:
                for i in range(0, total_rows, chunk_size):
                    sheet_name = f"Data_Part_{(i // chunk_size) + 1}"
                    chunk = full_df.slice(i, chunk_size)
                    
                    # This method is the most stable across all Polars versions
                    chunk.write_excel(workbook=workbook, worksheet=sheet_name)
            # ------------------------
            
            self.status_var.set(f"Excel Ready: {total_rows:,} rows.")
            messagebox.showinfo("Success", f"Created {total_rows:,} rows in {output_path}")
            
        except Exception as e: 
            messagebox.showerror("Final Error", f"Error Detail: {e}\n\nMake sure to run: pip install xlsxwriter")

    def open_full_file(self):
        path = os.path.join(self.base_path, self.full_excel_file)
        if os.path.exists(path): os.startfile(path)
        else: messagebox.showwarning("Missing", "Excel file not found.")

if __name__ == "__main__":
    root = tk.Tk()
    app = DataMasterPro(root)
    root.mainloop()
